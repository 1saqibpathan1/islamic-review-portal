وجہ سے وہ اس حالت کو پہنچے ہیں۔ یعنی علم کی وجہ سے وہ کوئی کاروباری نہیں بن سکے، پیسے والے نہیں بن سکے، اور آج وہ در در کی ٹھوکریں کھا رہے ہیں۔ اگر ان کی ضرورت پڑ جائے کسی کو قوم میں تو ایسے علماء کو فل چارج کرنا چاہیے، مکمل چارج کرنا چاہیے۔ وہ اتنا چارج کرنا چاہیے کہ کئی مہینے ان کے گھر کا خرچہ چلے، پورا چارج کرنا چاہیے ان کو۔ جیسے ایک مشہور روایت ہے کہ کوئی صحابہ کرام کہیں پہ جا رہے تھے تو کسی جگہ گئے، ان لوگوں نے ان کی ضیافت نہیں کی، حالانکہ ضیافت کرنا چاہیے۔ تو وہاں پہ کوئی شخص تھا، ان لوگوں کا کوئی سردار تھا، اس کو کسی جانور نے ڈنگ مار لیا۔ صحابہ کرام کے پاس آئے اور کہا کہ آپ رکیا کر دو، تو انہوں نے فوراً چارج کیا۔ حالانکہ ان صحابہ کرام کے بارے میں یہ نہیں ملتا کہ وہ جگہ جگہ جا کے رکیا کر کے پیسہ لے رہے ہیں۔ نہیں، لیکن یہاں پہ جب دیکھا کہ ان لوگوں نے ہمارے ساتھ رعایت نہیں کی، تو آج اگر انہیں ہماری ضرورت پڑ گئی ہے تو ہم بھی چارج کریں گے، فل چارج کیا ہوں گے۔ اتنا چارج کیا کہ اللہ کے نبی صلی اللہ علیہ وسلم کو جب پتہ چلا تو اللہ کے نبی صلی اللہ علیہ وسلم نے کہا کہ میرا بھی حصہ اس میں دو۔ اتنا وصول آیا صحابہ کرام نے۔ تو ایسے علماء کو بھی جم کر وصولنا چاہیے۔ جن علماء کی کوئی دیکھ بھال نہیں کر رہا ہے، اگر ان کی ضرورت پڑ جائے کسی کو تو پوری طرح وصولنا چاہیے، فل چارج کرنا چاہیے ان کو۔ تو میرے بھائی، اس طرح کے سوالات آپ اٹھا کے آپ کیا بتانا چاہتے ہیں لوگوں کو؟ آپ کیا میسج دینا چاہتے ہیں لوگوں کو؟ کہ علماء ان کے پاس پیٹ نہیں ہے، ان کے پاس परिवار نہیں ہے، ان کے پاس فیملی نہیں ہے، وہ ہوا پی کے رہتے ہیں، تو ان کے پاس کوئی کاروبار نہیں، کچھ نہیں ہے۔ تو آج اگر آپ ان کو بلا رہے ہیں، آپ کو ان کی ضرورت ہے، تو اپنے حساب سے وہ پیسے کا ڈیمانڈ کر رہے ہیں، تو اس میں کیا غلط ہے؟ اور کتنا بھی مانگ رہے ہیں، کتنا بھی مانگ رہے ہیں، پھر بھی ڈاکٹر، انجینئر، وکیل سے تو کم ہی مانگ رہے ہیں۔ حالانکہ اس سے زیادہ بھی مانگ لیں تو کوئی مسئلہ نہیں ہے۔ جس سماج میں، جس معاشرے میں ڈاکٹر کی فیس اتنی ہائی ہو، انجینئر کی فیس اتنی ہائی ہو، وکیل کی فیس اتنی ہائی ہو، اگر اس سماج میں ایک عالم ان سے بڑھ کر بھی مطالبہ کر لے تو کچھ غلط نہیں ہے۔ اس لیے میرے بھائی، آپ یہ دو فرق سمجھیں۔ ایک ہوتا ہے خود سے دین کا کام کرنا، کریں جتنا کرنا ہو، نو پرابلم، آپ کو تقریر کرنا ہے، عالم بنیے، تقریر کیجئے۔ اور دوسرا ہوتا ہے کسی کو ہائر کرنا۔ تو آپ اگر کسی کو ہائر کر رہے ہیں، یہ اس کی مرضی ہے، بھائی، یہ اس کا وقت ہے، یہ اس کا حق ہے، وہ اس کے لیے جو قیمت لگائے، یہ اس کا حق ہے، یہ اس کا اپنا مسئلہ ہے۔ آپ کو جمتا ہے تو بلائیے، نہیں جمتا تو مت بلائیے، لیکن آپ یہاں پہ فتویٰ مت لگائیے۔ ایسے ہی اور ویڈیوز بنانے میں
